#include <cstdlib>

#define TAMANHO 30

void inserir(float [], float , int &, int );
void mostrar(float [], int );
int percorrer(float [],float , int );

int main( )
{
    setlocale(LC_ALL,"Portuguese");
    float vetNotas[TAMANHO];
    int contaNotas = 0;

    float somaNotas=0, media;

    {

void inserir(float vet[], float notaAux, int &n, int tamanho) 
{
       if (tamanho == n)
          cout<< "ERRO: Lista cheia.";
       else
       {
           vet[n] = notaAux;
           n++;
       }
}
void mostrar (float vetNotasAux[], int tamAux)
{
     cout << "\n*** Notas da turma: ***";
     for (int i = 0; i < tamAux; i++)
     {
       cout << "\nNota " << i + 1 << ": " 
            <<vetNotasAux[i]; 
     }
}

int percorrer(float vetNotasAux[],float mediaAux, int tamAux)
{
    int notasAcima = 0;
    for (int i = 0; i < tamAux; i++)
    {
       if (vetNotasAux[i] > mediaAux)
          notasAcima ++; 
    }
    return  notasAcima ;         
}
